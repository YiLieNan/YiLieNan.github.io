<script>
function getUnitValue(id) {
  var v = parseFloat(document.getElementById(id).value);
  if (isNaN(v)) return null;
  var m = { 'V':1, 'mV':0.001, 'kV':1000, 'A':1, 'mA':0.001, 'µA':0.000001, 'Ω':1, 'kΩ':1000, 'MΩ':1000000 };
  var s = document.getElementById(id + '-unit');
  return s ? v * (m[s.value] || 1) : v;
}
function fmt(v) {
  if (v===null||isNaN(v)) return '—';
  if (Math.abs(v)>=1e6) return (v/1e6).toFixed(3)+'M';
  if (Math.abs(v)>=1e3) return (v/1e3).toFixed(3)+'k';
  if (Math.abs(v)>=1) return v.toFixed(3);
  if (Math.abs(v)>=1e-3) return (v*1e3).toFixed(3)+'m';
  return v.toExponential(3);
}
function fmtR(v) {
  if (v>=1e6) return (v/1e6).toFixed(2)+' MΩ';
  if (v>=1e3) return (v/1e3).toFixed(2)+' kΩ';
  return v.toFixed(2)+' Ω';
}
function showRes(id, html, tags) {
  var el = document.getElementById(id);
  el.innerHTML = html + (tags ? '<br>📖 '+tags.map(function(t){return '<span class=knowledge-tag>'+t+'</span>'}).join('') : '');
  el.classList.add('show');
}
function clearFields(p) {
  document.querySelectorAll('[id^="'+p+'-"]').forEach(function(e){if(e.tagName==='INPUT')e.value=''});
  var r = document.getElementById(p+'-result'); if(r) r.classList.remove('show');
}
function calcOhm() {
  var v=getUnitValue('ohm-v'), i=getUnitValue('ohm-i'), r=getUnitValue('ohm-r');
  var n=[v!==null,i!==null,r!==null].filter(Boolean).length;
  if(n<2){showRes('ohm-result','⚠️ 至少输入两个');return;}
  var h='',k=['欧姆定律 V=IR'];
  if(v===null){v=i*r;h+='V = '+fmt(i)+'×'+fmt(r)+' = <span class=highlight>'+fmt(v)+' V</span><br>';}
  if(i===null){i=v/r;h+='I = '+fmt(v)+'/'+fmt(r)+' = <span class=highlight>'+fmt(i)+' A</span><br>';}
  if(r===null){r=v/i;h+='R = '+fmt(v)+'/'+fmt(i)+' = <span class=highlight>'+fmtR(r)+'</span><br>';}
  showRes('ohm-result',h,k);
}
function calcSeries() {
  var r1=parseFloat(document.getElementById('series-r1').value);
  var r2=parseFloat(document.getElementById('series-r2').value);
  var v=parseFloat(document.getElementById('series-v').value);
  if(isNaN(r1)||isNaN(r2)||isNaN(v)){showRes('series-result','⚠️ 请填全');return;}
  var rt=r1+r2, i=v/rt, v1=i*r1, v2=i*r2;
  showRes('series-result','R总='+fmtR(rt)+'<br>I='+fmt(i)+' A<br>V₁='+fmt(v1)+' V<br>V₂='+fmt(v2)+' V',['串联R总=R₁+R₂','分压原理']);
}
function calcParallel() {
  var r1=parseFloat(document.getElementById('para-r1').value);
  var r2=parseFloat(document.getElementById('para-r2').value);
  var v=parseFloat(document.getElementById('para-v').value);
  if(isNaN(r1)||isNaN(r2)||isNaN(v)){showRes('para-result','⚠️ 请填全');return;}
  var rt=1/(1/r1+1/r2), i1=v/r1, i2=v/r2, it=i1+i2;
  showRes('para-result','R总='+fmtR(rt)+'<br>I₁='+fmt(i1)+' A<br>I₂='+fmt(i2)+' A<br>I总='+fmt(it)+' A',['并联1/R总=1/R₁+1/R₂','KCL']);
}
function calcDivider() {
  var vin=parseFloat(document.getElementById('div-vin').value);
  var r1=parseFloat(document.getElementById('div-r1').value);
  var r2=parseFloat(document.getElementById('div-r2').value);
  if(isNaN(vin)||isNaN(r1)||isNaN(r2)){showRes('div-result','⚠️ 请填全');return;}
  var vout=vin*r2/(r1+r2);
  showRes('div-result','Vout='+fmt(vin)+'×'+fmtR(r2)+'/'+fmtR(r1+r2)+'<br>=<span class=highlight>'+fmt(vout)+' V</span><br>分压比='+(r2/(r1+r2)*100).toFixed(1)+'%',['分压公式']);
}
function calcPower() {
  var v=parseFloat(document.getElementById('pow-v').value);
  var i=parseFloat(document.getElementById('pow-i').value);
  var r=parseFloat(document.getElementById('pow-r').value);
  var n=[!isNaN(v),!isNaN(i),!isNaN(r)].filter(Boolean).length;
  if(n<2){showRes('pow-result','⚠️ 至少输入两个');return;}
  var h='',k=['P=VI=I²R=V²/R'];
  if(!isNaN(v)&&!isNaN(i)){h='P='+fmt(v)+'×'+fmt(i)+'=<span class=highlight>'+fmt(v*i)+' W</span><br>';if(isNaN(r)){r=v/i;h+='R='+fmtR(r)+'<br>';k.push('欧姆定律');}}
  if(!isNaN(i)&&!isNaN(r)&&isNaN(v)){v=i*r;h+='V='+fmt(v)+' V<br>P=I²R='+fmt(i*i*r)+' W<br>';k.push('欧姆定律');}
  if(!isNaN(v)&&!isNaN(r)&&isNaN(i)){i=v/r;h+='I='+fmt(i)+' A<br>P=V²/R='+fmt(v*v/r)+' W<br>';k.push('欧姆定律');}
  showRes('pow-result',h,k);
}
function calcColorCode() {
  var d1=parseInt(document.getElementById('cc-1').value);
  var d2=parseInt(document.getElementById('cc-2').value);
  var mul=parseFloat(document.getElementById('cc-3').value);
  var err=document.getElementById('cc-4').value;
  var val=(d1*10+d2)*mul;
  showRes('cc-result','阻值=<span class=highlight>'+fmtR(val)+'</span><br>误差='+err,['色环读数']);
}
function calcRC() {
  var r=parseFloat(document.getElementById('rc-r').value);
  var c=parseFloat(document.getElementById('rc-c').value);
  if(isNaN(r)||isNaN(c)){showRes('rc-result','⚠️ 请填全');return;}
  var tau=r*c;
  showRes('rc-result','τ=R×C='+fmtR(r)+'×'+fmt(c)+'F<br>=<span class=highlight>'+fmt(tau)+' 秒</span><br>5τ≈'+fmt(tau*5)+' s 充满',['RC时间常数τ=RC']);
}
